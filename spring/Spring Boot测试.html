# 集成测试自动配置

Spring Framework的核心工作是将所有组件编织在一起，构成一个应用程序。整个过程就是：

  1. 读取配置说明（XML、基于Java的配置、基于Groovy的配置或其他类型的配置）。

  2. 在应用程序上下文里初始化Bean。

  3. 将Bean注入依赖它们的其他Bean中。

对Spring应用程序进行集成测试时，让Spring遵照生产环境来组装测试目标Bean。

从Spring 2.5开始，用SpringJUnit4ClassRunner来进行集成测试。它是一个JUnit类运行器，会为JUnit测试加载Spring应用程序上下文，并为测试类自动织入所需的Bean。这个类作为@RunWith注解的参数。

@RunWith(SpringJUnit4ClassRunner::class)
@ContextConfiguration(classes = [ReadingListApplication::class])
class ReadingListRepositoryTest {

    @Autowired
    private lateinit var repository: ReadingListRepository

    @Test
    fun testSaveBook() {
        val book = Book()
        book.reader = "daking"
        book.author = "daking"
        book.title = "book1"
        book.isbn = "isbn1"
        book.description = "desc1"

        repository.save(book)

        println(repository.findByReader("daking"))
    }
}

@ContextConfiguration指定了如何加载Spring应用程序上下文。这里是指定加载ReadingListApplication类配置的上下文。

可以用@SpringApplicationConfiguration来代替@ContextConfiguration。区别是：@ContextConfiguration没能加载完整的Spring Boot，而@SpringApplicationConfiguration可以。后者除了加载上下文，还会开启日志、加载外部属性（application.properties或application.yml）以及其他Spring Boot特性。

注意，从Spring Boot 1.4起，使用@SpringBootTest来代替@SpringApplicationConfiguration。

# 测试Web应用程序

Spring Boot测试Web应用程序有两个可选的方案：

  • Spring Mock MVC：能在一个近似真实的模拟Servlet容器中测试控制器，而不同实际启动应用服务器。

  • Web集成测试：在嵌入式Servlet容器（比如Tomcat、Jetty）中启动应用程序，在真正的应用服务器中执行测试。

# Spring Mock MVC

@SpringBootTest的webEnvironment参数默认为WebEnvironment.MOCK，即Spring Mock MVC环境。

@SpringBootTest可以使用classes参数来显式地指定要加载的应用程序上下文。若无显式指定，Spring Boot Test会自动搜索上下文：先找@Configuration，后找@SpringBootConfiguration。

package tech.daking.readinglist

import org.hamcrest.Matchers
import org.junit.Before
import org.junit.Test
import org.junit.runner.RunWith
import org.springframework.beans.factory.annotation.Autowired
import org.springframework.boot.test.context.SpringBootTest
import org.springframework.http.MediaType
import org.springframework.test.context.junit4.SpringJUnit4ClassRunner
import org.springframework.test.context.web.WebAppConfiguration
import org.springframework.test.web.servlet.MockMvc
import org.springframework.test.web.servlet.request.MockMvcRequestBuilders
import org.springframework.test.web.servlet.result.MockMvcResultMatchers
import org.springframework.test.web.servlet.setup.MockMvcBuilders
import org.springframework.web.context.WebApplicationContext
import tech.daking.readinglist.bean.Book

@RunWith(SpringJUnit4ClassRunner::class)
@SpringBootTest
class MockMvcWebTests {

    @Autowired
    private lateinit var webContext: WebApplicationContext

    private lateinit var mockMvc: MockMvc

    @Before
    fun setupMockMvc() {
        // 设置MockMvc
        mockMvc = MockMvcBuilders
                .webAppContextSetup(webContext)
                .build()
    }

    @Test
    fun homePage() {
        mockMvc.perform(MockMvcRequestBuilders.get("/readingList"))
                .andExpect(MockMvcResultMatchers.status().isOk)
                .andExpect(MockMvcResultMatchers.view().name("readingList"))
                .andExpect(MockMvcResultMatchers.model().attributeExists("books"))
                .andExpect(MockMvcResultMatchers.model().attribute<Collection<Book>>("books", Matchers.`is`(Matchers.empty())))
    }

    @Test
    fun postBook() {
        val expectedBook = Book()
        expectedBook.id = 1L
        expectedBook.reader = "readingList"
        expectedBook.title = "BOOK TITLE"
        expectedBook.author = "BOOK AUTHOR"
        expectedBook.isbn = "1234567890"
        expectedBook.description = "DESCRIPTION"

        mockMvc.perform(MockMvcRequestBuilders.post("/readingList")
                .contentType(MediaType.APPLICATION_FORM_URLENCODED)
                .param("title", "BOOK TITLE")
                .param("author", "BOOK AUTHOR")
                .param("isbn", "1234567890")
                .param("description", "DESCRIPTION"))
                .andExpect(MockMvcResultMatchers.status().is3xxRedirection)
                .andExpect(MockMvcResultMatchers.header().string("Location", "/readingList"))

        mockMvc.perform(MockMvcRequestBuilders.get("/readingList"))
                .andExpect(MockMvcResultMatchers.status().isOk)
                .andExpect(MockMvcResultMatchers.view().name("readingList"))
                .andExpect(MockMvcResultMatchers.model().attributeExists("books"))
                .andExpect(MockMvcResultMatchers.model().attribute("books",
                        Matchers.hasSize<Collection<Book>>(1)))
                .andExpect(MockMvcResultMatchers.model().attribute("books",
                        Matchers.contains(Matchers.samePropertyValuesAs(expectedBook))))

    }
}

# 固定端口的服务器

端口默认为8080，可以通过以下方式来指定端口：（根据优先级排序）

  1. 在属性文件中指定，比如application.properties文件中的server.port=8000。

  2. 在@SpringBootTest中使用properties参数来指定，比如properties = ["server.port=8000"]。

@SpringBootTest的webEnvironment参数要设置为WebEnvironment.DEFINED_PORT。

package tech.daking.readinglist

import org.junit.Assert
import org.junit.Test
import org.junit.runner.RunWith
import org.springframework.boot.test.context.SpringBootTest
import org.springframework.http.HttpStatus
import org.springframework.test.context.junit4.SpringJUnit4ClassRunner
import org.springframework.web.client.HttpClientErrorException
import org.springframework.web.client.RestTemplate

@RunWith(SpringJUnit4ClassRunner::class)
@SpringBootTest(webEnvironment = SpringBootTest.WebEnvironment.DEFINED_PORT) // 固定端口
class DefinedPortWebTest {

    @Test(expected = HttpClientErrorException::class)
    fun pageNotFound() {
        try {
            val rest = RestTemplate()
            rest.getForObject("http://localhost:8000/list/bogusPage", String::class.java)
        } catch (e: HttpClientErrorException) {
            Assert.assertEquals(HttpStatus.NOT_FOUND, e.statusCode)
            throw e
        }
    }

    @Test
    fun pageFound() {
        val rest = RestTemplate()
        val resp = rest.getForObject("http://localhost:8000/readingList", String::class.java)
        Assert.assertNotNull(resp)
    }
}

# 随机端口的服务器

@SpringBootTest的webEnvironment参数要设置为WebEnvironment.RANDOM_PORT。

@RunWith(SpringJUnit4ClassRunner::class)
@SpringBootTest(webEnvironment = SpringBootTest.WebEnvironment.RANDOM_PORT) // 随机端口
class RandomPortWebTest {

    // 真实端口值注入
    @Value("\${local.server.port}")
    private var port: Int = 0

    @Test(expected = HttpClientErrorException::class)
    fun pageNotFound() {
        try {
            val rest = RestTemplate()
            rest.getForObject("http://localhost:$port/list/bogusPage", String::class.java)
        } catch (e: HttpClientErrorException) {
            Assert.assertEquals(HttpStatus.NOT_FOUND, e.statusCode)
            throw e
        }
    }

    @Test
    fun pageFound() {
        val rest = RestTemplate()
        val resp = rest.getForObject("http://localhost:$port/readingList", String::class.java)
        Assert.assertNotNull(resp)
    }
}
